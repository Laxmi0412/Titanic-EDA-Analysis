             color='steelblue', edgecolor='white')
axes[0].set_title('Age Distribution')
axes[0].set_xlabel('Age')

df[df['survived']==1]['age'].dropna().hist(
    bins=30, alpha=0.6, color='green',
    label='Survived', ax=axes[1])
df[df['survived']==0]['age'].dropna().hist(
    bins=30, alpha=0.6, color='red',
    label='Did not survive', ax=axes[1])
axes[1].set_title('Age Distribution by Survival')
axes[1].set_xlabel('Age')
axes[1].legend()

plt.tight_layout()
plt.show()
